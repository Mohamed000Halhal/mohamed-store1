rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is admin
    function isAdmin() {
      return request.auth != null && (
        request.auth.token.email == 'mohamed@store.com' || 
        request.auth.token.email == 'mohamedemara@store.com' || 
        request.auth.token.email == 'mohamedemara@stroe.com'
      );
    }

    // --- Public Access (Read-Only) ---

    // Products: Everyone can browse products
    match /products/{productId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Settings (UI/Theme): Everyone needs to see the store design theme
    match /settings/{settingId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // --- Store Operations ---

    // Orders: Customers can create orders. Admin & Assigned Secretary can manage.
    match /orders/{orderId} {
      allow create: if true;
      allow read, update: if isAdmin() || (request.auth != null && (request.auth.uid == resource.data.secretaryId || request.auth.token.email == resource.data.secretaryEmail));
      allow delete: if isAdmin();
    }
    
    // Contact Messages: Customers can send messages.
    match /contacts/{contactId} {
      allow create: if true;
      allow read, delete: if isAdmin();
      allow update: if false;
    }

    // --- Administrative Area (Admin Only) ---
    
    match /secretaries/{secretaryId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
      allow update: if request.auth != null && request.auth.token.email == resource.data.email;
    }
    
    match /deletedOrders/{orderId} {
      allow read, write: if isAdmin();
    }

    match /adminTokens/{tokenId} {
      allow read, write: if isAdmin();
    }
    
    match /notifications/{notificationId} {
      allow read, write: if isAdmin();
    }
    
    match /adminStats/{document=**} {
      allow read, write: if isAdmin();
    }
    
    match /adminPresence/{document=**} {
      allow read, write: if isAdmin();
    }
    
    match /adminSessionLog/{document=**} {
      allow read, write: if isAdmin();
    }

    // --- Personal User Data ---

    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    match /carts/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
